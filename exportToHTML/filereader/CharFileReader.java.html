13   public class CharFileReader implements IFileReader, ICharFileReaderConstruction { 
14       Charset encoding = Charset.defaultCharset(); 
15       private File file; 
16       private BufferedReader reader; 
17    
18       // construction access 
19       public ICharFileReaderConstruction getConstructor() { 
20           return this; 
21       } 
22    
23       public void setFilePath(String path) { 
24           this.file = new File(path); 
25       } 
26    
27       public void setupFile() { 
28           try { 
29               InputStream in = new FileInputStream(this.file); 
30               Reader inputReader = new InputStreamReader(in, this.encoding); 
31               this.reader = new BufferedReader(inputReader); 
32           } catch (FileNotFoundException e) { 
33               System.out.println("CharFileReader - FileNotFoundException : File :<" + this.file + "> not found "); 
34               System.exit(1); 
35           } 
36       } 
37    
38       @Override 
39       public String nextChar(ICharFileReaderCallback fileReadCallback) { 
40           String nextCharInFile = null; 
41           try { 
42               int nextInt = this.reader.read(); 
43               NumberUtil.requirePositive(nextInt); 
44               fileReadCallback.success(String.valueOf((char) nextInt)); 
45           } catch (IOException e) { 
46               System.out.println("CharFileReader - IOException : File :<" + this.file + ">  "); 
47               System.exit(1); 
48           } catch (Exception e) { 
49               fileReadCallback.failed(); 
50           } 
51           return nextCharInFile; 
52       } 
53   }
