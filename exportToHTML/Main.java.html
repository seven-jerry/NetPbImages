1    import processing.build.PBMImageBuildDelegate;

10   import java.util.ArrayList; 
11   import java.util.List; 
12    
13           /* 
14            * this program reads in pbm images(p1,p2,p3) 
15            * and applies rotation and color inversion 
16            * 
17            * it uses xml files that define the structure of the file 
18            * 
19            * java -jar pbm-tool <input-file> -operation1 -operation2 -operation3 <output-file> 
20            * */ 
21    
22   public class Main { 
23       public static void main(String[] args) { 
24    
25           // validate arguments 
26           String inputFilePath = Main.getInputFilePathArg(args); 
27           String outputFilePath = Main.getOutputFilePathArg(args); 
28           List<String> actions = Main.getActionParamArgs(args); 
29    
30           //initialization 
31           CharFileReader reader = new CharFileReader(); 
32           PreSectionConfiguration configFile = new PreSectionConfiguration(); 
33           ContentConfiguration sectionConfig = new ContentConfiguration(); 
34           PBMImageBuilder imageBuilder = PBMImageBuilder.builder(); 
35    
36           // setup objects 
37           reader.getConstructor().setFilePath(inputFilePath); 
38           imageBuilder.getConstructor().setPreSectionLoader(configFile); 
39           imageBuilder.getConstructor().setContentSectionLoader(sectionConfig); 
40           imageBuilder.getConstructor().setFileReader(reader); 
41           imageBuilder.getConstructor().setConfigFilePath("pbm_header_format_config.xml"); 
42           imageBuilder.getConstructor().setBuildDelegate(new PBMImageBuildDelegate()); 
43           imageBuilder.getConstructor().setup(); 
44           imageBuilder.getConstructor().execute(); 
45    
46           System.out.println("input :\n"+imageBuilder.getTargetImage()); 
47           // apply actions 
48           Main.callActions(imageBuilder.getTargetImageDelegate(), actions); 
49           PBMImageFileWriter.writeToFile(PBMImageBuilder.builder().getTargetImage(), outputFilePath); 
50           System.out.println("output :\n"+imageBuilder.getTargetImage()); 
51    
52       } 
53    
54       // the first param is expected to be the source image file path 
55       private static String getInputFilePathArg(final String[] args) { 
56           try { 
57               String inputFile = args[0]; 
58               return inputFile; 
59           } catch (Exception e) { 
60               System.out.println("the first param must be the input file"); 
61               System.out.println("java -jar pbm-tool <input-file> -operation1 -operation2 -operation3 <output-file>"); 
62               System.exit(-1); 
63           } 
64           return null; 
65       } 
66    
67       // the last param is expected to be the destination image file path 
68       private static String getOutputFilePathArg(final String[] args) { 
69           try { 
70               ArrayUtil.requireLength(args,2); 
71               String outFile = args[args.length - 1]; 
72               return outFile; 
73           } catch (Exception e) { 
74               System.out.println("the last param must be the output file"); 
75               System.out.println("java -jar pbm-tool <input-file> -operation1 -operation2 -operation3 <output-file>"); 
76               System.exit(-1); 
77           } 
78           return null; 
79       } 
80    
81       // load the actions (invert,rotate) into a array 
82       // operations are expected to be between inputfile(firstElement) and outputfile(lastElement) 
83       private static List<String> getActionParamArgs(final String[] args) { 
84           List<String> actions = new ArrayList<>(); 
85           try { 
86    
87               for (int i = 1; i < args.length - 1; i++) { 
88                   actions.add(args[i]); 
89               } 
90           } catch (Exception e) { 
91               System.out.println("the last param must be the output file"); 
92               System.out.println("java -jar pbm-tool <input-file> -operation1 -operation2 -operation3 <output-file>"); 
93               System.exit(-1); 
94           } 
95           return actions; 
96       } 
97    
98       private static void callActions(Object targetObject, List<String> actions) { 
99           for (String action : actions) { 
100              ReflectionUtil.callValueAnnotatedMethod(targetObject, action); 
101          } 
102      }
