18       @Override 
19       public void rotate() { 
20           LinkedList<Integer> newBodyValues = new LinkedList<>(); 
21           Collections.reverse(this.bodyValues); 
22           Iterator<Integer> it = bodyValues.iterator(); 
23           while (it.hasNext()) { 
24               //get colors in reverse order 
25               Integer b = it.next(); 
26               Integer g = it.next(); 
27               Integer r = it.next(); 
28               //put them into order 
29               newBodyValues.add(r); 
30               newBodyValues.add(g); 
31               newBodyValues.add(b); 
32           } 
33           this.bodyValues = newBodyValues; 
34       } 
35    
36       @Override 
37       public String toString() { 
38           String imageString = this.getSubtypeName() + "\n"; 
39           for (String comment : this.comment) { 
40               imageString += comment + "\n"; 
41           } 
42           imageString += this.width + " " + this.height + "\n"; 
43           imageString += this.maxColor + "\n"; 
44    
45           for (Integer bodyValue : this.bodyValues) { 
46               imageString += bodyValue; 
47               imageString += "\n"; 
48    
49           } 
50           return imageString; 
51       }
