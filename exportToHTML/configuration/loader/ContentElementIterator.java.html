13   public class ContentElementIterator implements Iterator<Element> { 
14       private int currentIndex; 
15       private List<Node> childNodes = new ArrayList<>(); 
16    
17       public void setParentElement(Element parent) { 
18           NodeList children = parent.getChildNodes(); 
19           this.childNodes = new ArrayList<>(); 
20           for (int i = 0; i < children.getLength(); ++i) { 
21               Element child = (Element) children.item(i); 
22               this.childNodes.add(child); 
23           } 
24           this.currentIndex = 0; 
25       } 
26    
27       public void addNode(Node node) { 
28           this.childNodes.add(node); 
29       } 
30    
31       @Override 
32       public boolean hasNext() { 
33           return this.currentIndex < this.childNodes.size(); 
34       } 
35    
36       @Override 
37       public Element next() { 
38           Node nextElement = this.childNodes.get(this.currentIndex); 
39           this.currentIndex++; 
40           return (Element) nextElement; 
41       }
