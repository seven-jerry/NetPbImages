20   /* 
21    * this loads the configuration for the pre section 
22    * 
23    * */ 
24   public class PreSectionConfiguration implements IPreSectionLoader { 
25       private File configFile; 
26    
27    
28       @Override 
29       public void setup() { 
30           Objects.requireNonNull(this.configFile, "config file must not be null"); 
31           this.setupPreSection(); 
32       } 
33    
34       public void setConfigFile(File configFile) { 
35           this.configFile = configFile; 
36       } 
37    
38       private final void setupPreSection() { 
39           try { 
40               DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance(); 
41               DocumentBuilder dBuilder = dbFactory.newDocumentBuilder(); 
42               Document doc = dBuilder.parse(this.configFile); 
43               doc.normalize(); 
44               DomUtil.trimWhitespace(doc.getDocumentElement()); 
45               Element preElement = (Element) doc.getDocumentElement().getElementsByTagName("pre").item(0); 
46               //setup target keys 
47               Element targetKeys = (Element) preElement.getElementsByTagName("keys").item(0); 
48               this.setupPreTargetKeys(targetKeys.getElementsByTagName("key")); 
49               //setup variables 
50               Element variables = (Element) preElement.getElementsByTagName("variables").item(0); 
51               this.setupPreVariables(variables.getElementsByTagName("Char")); 
52               this.setupSystemVariables(); 
53               //setup procedures 
54               Element procedures = (Element) preElement.getElementsByTagName("procedures").item(0); 
55               this.setupProcedures(procedures.getElementsByTagName("Procedure")); 
56    
57           } catch (Exception e) { 
58               System.out.println("ContentConfiguration - setup : " + e.getMessage()); 
59               System.exit(-1); 
60           } 
61    
62       } 
63    
64       /* Setup Target Keys */ 
65       private final void setupPreTargetKeys(final NodeList keys) throws ParserConfigurationException, SAXException { 
66           for (int i = 0; i < keys.getLength(); i++) { 
67               Element key = (Element) keys.item(i); 
68               String docId = key.getAttribute("docId"); 
69               String targetKey = key.getAttribute("targetKey"); 
70               this.addTargetKey(docId, targetKey); 
71           } 
72       } 
73    
74       private final void addTargetKey(final String docId, final String targetKey) { 
75           try { 
76               Objects.requireNonNull(docId); 
77               Objects.requireNonNull(targetKey); 
78               this.getImageBuilder().putTargetKey(docId, TargetKey.getEnum(targetKey)); 
79           } catch (NullPointerException ex) { 
80               System.out.println("Failed to add target key: docId:<" + docId + "> targetKey:<" + targetKey + ">"); 
81           } catch (IllegalArgumentException ex) { 
82               System.out.println("Could not convert targetKey into enum key. givenKey:<" + targetKey + ">"); 
83           } 
84       } 
85    
86       /* Setup Variables */ 
87       private final void setupPreVariables(final NodeList variables) { 
88           for (int i = 0; i < variables.getLength(); i++) { 
89               Element var = (Element) variables.item(i); 
90               String docId = var.getAttribute("docId"); 
91               String value = var.getAttribute("range"); 
92               this.addVariable(docId, value); 
93           } 
94       } 
95    
96       private void setupSystemVariables() { 
97           this.addVariable("whitespace", CharUtil.whitespace()); 
98           this.addVariable("maxIntegerValue", Integer.valueOf(Integer.MAX_VALUE).toString()); 
99    
100      } 
101   
102      private final void addVariable(final String docId, final String value) { 
103          try { 
104              Objects.requireNonNull(docId); 
105              Objects.requireNonNull(value); 
106              this.getImageBuilder().putVariable(docId, new CharType(value)); 
107          } catch (NullPointerException ex) { 
108              System.out.println("NullPointerException - Failed to add variable key: docId:<" + docId + "> value:<" + value + ">"); 
109          } catch (IllegalArgumentException ex) { 
110              System.out.println("IllegalArgumentException - Could not convert variable into chartype. value:<" + value + ">"); 
111          } 
112      } 
113   
114      /* Setup Procedures */ 
115      private final void setupProcedures(final NodeList procedures) { 
116          for (int i = 0; i < procedures.getLength(); i++) { 
117              Element procedure = (Element) procedures.item(i); 
118              String docId = procedure.getAttribute("docId"); 
119              String className = procedure.getAttribute("className"); 
120              this.addProcedure(docId, className, procedure); 
121          } 
122      } 
123      private final void addProcedure(final String docId, final String className, final Element procedureElement) { 
124          try { 
125              Objects.requireNonNull(docId); 
126              Objects.requireNonNull(className); 
127   
128              Class<?> procedureClass = Class.forName("processing.procedure." + className + "Procedure"); 
129              AbstractProcedure procedure = (AbstractProcedure) procedureClass.newInstance(); 
130              procedure.setDocId(docId); 
131              procedure.proccessXMLElement(procedureElement); 
132              PBMImageBuilder.builder().putProcedure(docId, procedure); 
133          } catch (NullPointerException ex) { 
134              System.out.println("NullPointerException - Failed to add procedure: docId:<" + docId + "> className:<" + className + ">"); 
135          } catch (IllegalArgumentException ex) { 
136              System.out.println("IllegalArgumentException - Could not convert className into procedure. className:<" + className + ">"); 
137          } catch (ClassNotFoundException ex) { 
138              System.out.println("ClassNotFoundException - Could not convert procedure into class. className:<" + className + ">"); 
139          } catch (IllegalAccessException e) { 
140              e.printStackTrace(); 
141          } catch (InstantiationException e) { 
142              e.printStackTrace(); 
143          } 
144      } 
145   
146      private IPreSectionAccess getImageBuilder() { 
147          return PBMImageBuilder.builder(); 
148      }
