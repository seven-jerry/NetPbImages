15   public class ContentConfiguration implements IContentSectionLoader { 
16       private IFileReader inputFileReader; 
17       private ContentElementIterator contentIterator; 
18       private File configFile; 
19    
20       public void setConfigFile(File configFile) { 
21           this.configFile = configFile; 
22       } 
23    
24       public void setInputFileReader(IFileReader inputFileReader) { 
25           this.inputFileReader = inputFileReader; 
26       } 
27    
28       @Override 
29       public void setup() { 
30           try { 
31               DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance(); 
32               DocumentBuilder dBuilder = dbFactory.newDocumentBuilder(); 
33               Document doc = dBuilder.parse(this.configFile); 
34               doc.normalize(); 
35               DomUtil.trimWhitespace(doc.getDocumentElement()); 
36    
37               Element contentElement = (Element) doc.getDocumentElement().getElementsByTagName("content").item(0); 
38               DomUtil.trimWhitespace(contentElement); 
39               this.contentIterator = new ContentElementIterator(); 
40               this.contentIterator.setParentElement(contentElement); 
41           } catch (Exception e) { 
42               System.out.println("ContentConfiguration - setup : " + e.getMessage()); 
43               System.exit(-1); 
44           } 
45       } 
46    
47    
48       @Override 
49       public void nextCharValidator() { 
50           while (this.contentIterator.hasNext() && PBMImageBuilder.builder().hasCharValidator() == false) { 
51               Element nextContentElement = this.contentIterator.next(); 
52               try { 
53                   IContentType element = (IContentType) Class.forName("configuration.datatype." + nextContentElement.getTagName() + "Type").newInstance(); 
54                   element.loadAsContentType(nextContentElement); 
55               } catch (Exception e) { 
56                   System.out.println("ContentConfiguration - nextCharValidator : " + e.getMessage()); 
57                   System.exit(-1); 
58               } 
59           } 
60       } 
61    
62       @Override 
63       public void addElement(Element element) { 
64           this.contentIterator.addNode(element); 
65       }
