1    package processing.datatype;

3    import processing.build.*; 
4    import processing.validation.ICharValidator;

11   /** 
12    * this class validates a certain character in a given range 
13    * this should be the last validator as it tell the cursor to load the nex Validation list 
14    * this will update the content for the provided docId 
15    */ 
16    
17   public class UpdateNumberType implements ICharContentProccessor, ICharValidator { 
18       private String docId; 
19       private String breakDocId; 
20       private String rangeDocId; 
21       private String content = ""; 
22    
23       private ICursor cursor; 
24    
25       private ICharCompareCallback callback = new CompareCallback(); 
26    
27       public UpdateNumberType(String docId, String rangeDocId, String breakDocId) { 
28           this.docId = docId; 
29           this.rangeDocId = rangeDocId; 
30           this.breakDocId = breakDocId; 
31       } 
32    
33       @Override 
34       public void proccesChar(ICursor cursor, String nextCharacter) { 
35           this.cursor = cursor; 
36           String range = PBMImageBuilder.builder().getVariableContent(this.breakDocId); 
37           CharUtil.inRange(nextCharacter, range, this.callback); 
38       } 
39    
40       private void validateNumber() { 
41           try { 
42               String rangeVarString = PBMImageBuilder.builder().getVariableContent(this.rangeDocId); 
43               Integer rangeVar = Integer.parseInt(rangeVarString); 
44               Integer number = Integer.parseInt(this.content); 
45               NumberUtil.smallerOrEqual(number,rangeVar,"info : UpdateNumberType - validateNumber : Number is greater that defined range"); 
46               PBMImageBuilder.builder().updateDocId(this.docId, number.toString()); 
47           } catch (Exception e) { 
48               System.out.println("unable to convert <" + this.content + "> into a integer "); 
49               PBMImageBuilder.builder().validationError(); 
50           } 
51       } 
52    
53       @Override 
54       public void charInRange(String needle, String docId, String range, ICharCompareCallback callback) { 
55           CharUtil.inRange(needle, range, callback); 
56       } 
57    
58       class CompareCallback implements ICharCompareCallback { 
59    
60           @Override 
61           public void success(String content) { 
62               UpdateNumberType.this.validateNumber(); 
63               UpdateNumberType.this.cursor.nextValidatorList(); 
64               UpdateNumberType.this.cursor.performChanges(); 
65           } 
66    
67           @Override 
68           public void fail(String content) { 
69               UpdateNumberType.this.content += content; 
70               UpdateNumberType.this.cursor.nextChar(); 
71               UpdateNumberType.this.cursor.performChanges(); 
72           } 
73       } 
74   }
