10   /** 
11    * this class validates a certain character in a given range 
12    * this should be the last validator as it tell the cursor to load the nex Validation list 
13    * this will update the content for the provided docId 
14    */ 
15   public class UpdateCharType implements ICharContentProccessor, ICharCompareCallback, ICharValidator { 
16       private String range; 
17       private String docId; 
18       private ICursor cursor; 
19    
20       public UpdateCharType(String docId, String rangeString) { 
21           this.range = rangeString; 
22           this.docId = docId; 
23       } 
24    
25       @Override 
26       public void proccesChar(ICursor cursor, String nextCharacter) { 
27           this.cursor = cursor; 
28           CharUtil.inRange(nextCharacter, range, this); 
29       } 
30    
31       public void proccesChar(String nextCharacter, ICharCompareCallback callback) { 
32           CharUtil.inRange(nextCharacter, range, callback); 
33       } 
34    
35       public String getString() { 
36           return this.range; 
37       } 
38    
39       @Override 
40       public void success(String content) { 
41           PBMImageBuilder.builder().updateDocId(this.docId, content); 
42           this.cursor.nextValidatorList(); 
43           this.cursor.nextChar(); 
44           this.cursor.performChanges(); 
45       } 
46    
47       @Override 
48       public void fail(String content) { 
49           System.out.println("Validation failed on UpdateCahrType range:<" + this.range + "< content:<" + content + ">"); 
50           PBMImageBuilder.builder().validationError(); 
51    
52       } 
53    
54       @Override 
55       public void charInRange(String needle, String docId, String range, ICharCompareCallback callback) { 
56           CharUtil.inRange(needle, range, callback); 
57       } 
58   }
