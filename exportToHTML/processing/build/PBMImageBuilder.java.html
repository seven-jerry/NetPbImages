207      private void setupContentIterator() { 
208          this.charContentList = new ArrayList<ICharContentProccessor>(); 
209          this.charContentList.addAll(procedures.values()); 
210          this.charContentList.addAll(subcontent.values()); 
211          this.charContentList.add(this.contentValidator); 
212          this.charContentList.add(this); 
213   
214          //   System.out.println(this.charContentList); 
215          this.charContentIterator = this.charContentList.iterator(); 
216      } 
217   
218      private void handleFileReadError() { 
219          this.cursor.currentChar = ""; 
220          this.cursor.currentContentIterationValidator = new EmptyCharContentProccessor(); 
221          this.contentValidator = null; 
222      }

224      class ContentReadCursor implements ICursor { 
225          private String currentChar; 
226          private ICharContentProccessor currentContentIterationValidator; 
227   
228          @Override 
229          public void nextChar() { 
230              PBMImageBuilder.this.fileReader.nextChar(PBMImageBuilder.this.fileReadCallback); 
231          } 
232   
233          @Override 
234          public void nextValidator() { 
235              this.currentContentIterationValidator = PBMImageBuilder.this.charContentIterator.next(); 
236          } 
237   
238          @Override 
239          public void nextValidatorList() { 
240              PBMImageBuilder.this.contentValidator = null; 
241              PBMImageBuilder.this.contentSectionLoader.nextCharValidator(); 
242              PBMImageBuilder.this.setupContentIterator(); 
243              this.currentContentIterationValidator = PBMImageBuilder.this.charContentIterator.next(); 
244          } 
245   
246          @Override 
247          public void performChanges() { 
248              this.currentContentIterationValidator.proccesChar(this, this.currentChar); 
249          } 
250   
251          @Override 
252          public void fillValidationList() { 
253              PBMImageBuilder.this.setupContentIterator(); 
254              this.currentContentIterationValidator = PBMImageBuilder.this.charContentIterator.next(); 
255          } 
256   
257          private void nextCharFromReader(String readChar) { 
258              PBMImageBuilder.this.debugString += readChar; 
259              this.currentChar = readChar; 
260          } 
261      }
