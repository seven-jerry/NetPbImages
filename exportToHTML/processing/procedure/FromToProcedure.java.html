12           /* 
13            * this class takes all strings from one char to another and proccesses them

17   public class FromToProcedure extends AbstractProcedure implements ICharContentProccessor { 
18       private ICharValidator fromChar; 
19       private String docId; 
20       String fromDocID; 
21       String fromRange; 
22       String content; 
23    
24       private ICharValidator toChar; 
25       String toDocId; 
26       String toRange; 
27    
28       private ICharValidator continuesWith; 
29       String continuesWithDocId; 
30       String continuesWithRange; 
31    
32       private ICharValidator next; 
33       String nextDocID; 
34       String nextRange; 
35       ICharCompareCallback nextCallback; 
36    
37       private FromCallback fromCallback = new FromCallback(); 
38       private ToCallback toCallback = new ToCallback(); 
39    
40       @Override 
41       public void proccessXMLElement(final Element procedureElement) { 
42           this.docId = procedureElement.getAttribute("docId"); 
43           Element from = (Element) procedureElement.getElementsByTagName("from").item(0); 
44           Element to = (Element) procedureElement.getElementsByTagName("to").item(0); 
45    
46           this.processFromElement(from); 
47           this.processToElement(to); 
48    
49           this.next = this.fromChar; 
50           this.nextDocID = this.fromDocID; 
51           this.nextRange = this.fromRange; 
52           this.nextCallback = this.fromCallback; 
53       } 
54    
55       @Override 
56       public void proccesChar(ICursor cursor, String nextCharacter) { 
57           super.proccesChar(cursor, nextCharacter); 
58           this.next.charInRange(nextCharacter, this.nextDocID, this.nextRange, this.nextCallback); 
59       } 
60    
61    
62       private final void processFromElement(final Element from) { 
63           try { 
64               Objects.requireNonNull(from); 
65               DomUtil.trimWhitespace(from); 
66               Element fromChild = (Element) from.getFirstChild(); 
67               String tagName = fromChild.getTagName(); 
68               this.fromChar = this.objectFromElementName(tagName); 
69               this.fromDocID = fromChild.getAttribute("docId"); 
70               this.fromRange = fromChild.getAttribute("range"); 
71    
72           } catch (NullPointerException ex) { 
73               System.out.println("NullPointerException - processFromElement"); 
74           } catch (IllegalArgumentException ex) { 
75               System.out.println("IllegalArgumentException - processFromElement"); 
76           } 
77       } 
78    
79       private final void processToElement(final Element to) { 
80           try { 
81               Objects.requireNonNull(to); 
82               DomUtil.trimWhitespace(to); 
83               Element fromChild = (Element) to.getFirstChild(); 
84               String tagName = fromChild.getTagName(); 
85               this.toChar = this.objectFromElementName(tagName); 
86               this.toDocId = fromChild.getAttribute("docId"); 
87               this.toRange = fromChild.getAttribute("range"); 
88    
89           } catch (NullPointerException ex) { 
90               System.out.println("NullPointerException - processToElement"); 
91           } catch (IllegalArgumentException ex) { 
92               System.out.println("IllegalArgumentException - processToElement"); 
93           } 
94       } 
95    
96       private final ICharValidator objectFromElementName(String elementName) { 
97           try { 
98               Objects.requireNonNull(elementName); 
99               Class characterAccessClass = Class.forName("configuration.datatype." + elementName + "Type"); 
100              return (ICharValidator) characterAccessClass.newInstance(); 
101          } catch (NullPointerException ex) { 
102              System.out.println("FromToProcedure - objectFromElementName : NullPointerException"); 
103          } catch (IllegalArgumentException ex) { 
104              System.out.println("FromToProcedure - objectFromElementName : IllegalArgumentException"); 
105          } catch (ClassNotFoundException ex) { 
106              System.out.println("FromToProcedure : objectFromELementName : Class <processing.contenttype.." + elementName + "Type> not found "); 
107          } catch (IllegalAccessException e) { 
108              System.out.println("FromToProcedure - IllegalAccessException"); 
109          } catch (InstantiationException e) { 
110              System.out.println("FromToProcedure - InstantiationException"); 
111          } 
112          return null; 
113      }

115      class FromCallback implements ICharCompareCallback { 
116   
117          @Override 
118          public void success(String content) { 
119   
120              FromToProcedure.this.content = content; 
121              FromToProcedure.this.next = FromToProcedure.this.toChar; 
122              FromToProcedure.this.nextDocID = FromToProcedure.this.toDocId; 
123              FromToProcedure.this.nextRange = FromToProcedure.this.toRange; 
124              FromToProcedure.this.nextCallback = FromToProcedure.this.toCallback; 
125              FromToProcedure.this.cursor.nextChar(); 
126              FromToProcedure.this.cursor.performChanges(); 
127          } 
128   
129          @Override 
130          public void fail(String content) { 
131              FromToProcedure.this.cursor.nextValidator(); 
132              FromToProcedure.this.cursor.performChanges(); 
133   
134          } 
135      }

137      class ToCallback implements ICharCompareCallback { 
138   
139   
140          @Override 
141          public void success(String content) { 
142   
143              FromToProcedure.this.next = FromToProcedure.this.fromChar; 
144              FromToProcedure.this.nextDocID = FromToProcedure.this.fromDocID; 
145              FromToProcedure.this.nextRange = FromToProcedure.this.fromRange; 
146              FromToProcedure.this.nextCallback = FromToProcedure.this.fromCallback; 
147   
148              PBMImageBuilder.builder().updateDocId(FromToProcedure.this.docId, FromToProcedure.this.content); 
149              FromToProcedure.this.content = ""; 
150              FromToProcedure.this.cursor.nextChar(); 
151              FromToProcedure.this.cursor.nextValidator(); 
152              FromToProcedure.this.cursor.performChanges(); 
153          } 
154   
155          @Override 
156          public void fail(String content) { 
157              FromToProcedure.this.content += content; 
158              FromToProcedure.this.cursor.nextChar(); 
159              FromToProcedure.this.cursor.performChanges(); 
160          } 
161      }
