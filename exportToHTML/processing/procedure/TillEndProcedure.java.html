14   public class TillEndProcedure extends AbstractProcedure {

17       @Override 
18       public void proccessXMLElement(final Element procedureElement) { 
19           DomUtil.trimWhitespace(procedureElement); 
20           children = procedureElement.getChildNodes(); 
21    
22           for (int i = 0; i < children.getLength(); ++i) { 
23               Element child = (Element) children.item(i); 
24               PBMImageBuilder.builder().addContentSectionElement(child); 
25           } 
26           PBMImageBuilder.builder().putProcedure(procedureElement.getAttribute("docId"), this); 
27       } 
28    
29       @Override 
30       public void proccesChar(ICursor cursor, String nextCharInFile) { 
31           for (int i = 0; i < children.getLength(); i++) { 
32               Element child = (Element) children.item(i); 
33               PBMImageBuilder.builder().addContentSectionElement(child); 
34           } 
35           cursor.nextValidator(); 
36           cursor.performChanges(); 
37       }
