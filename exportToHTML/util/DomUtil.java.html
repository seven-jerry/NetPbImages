8        public static void trimWhitespace(Node node) 
9        { 
10           NodeList children = node.getChildNodes(); 
11           for(int i = 0; i < children.getLength(); ++i) { 
12               Node child = children.item(i); 
13               if (child.getNodeType() == Node.TEXT_NODE) { 
14                   child.getParentNode().removeChild(child); 
15               } 
16               trimWhitespace(child); 
17           } 
18       }
